import streamlit as st import pandas as pd import plotly.express as px # הגדרות עמוד ותמיכה בכתיבה מימין לשמאל st.set_page_config(page_title="דשבורד קהילת רכזי הערכה תשפ\"ו", layout="wide") # הזרקת CSS להתאמת כיוון הטקסט לעברית (RTL) st.markdown(""" """, unsafe_allow_html=True) # כותרת ראשית st.title("📊 דשבורד מנהלים אינטראקטיבי - קהילת רכזי הערכה תשפ\"ו") st.subheader("ניתוח ממצאי משוב מסכם לצוות הפיקוח, המדריכים והמנהלים") st.markdown("---") # נתונים כמותיים מתוך המשוב data_quantitative = { "מדד הערכה": [ "שביעות רצון כללית מהקהילה", "רלוונטיות הנושאים לצרכי הרכזים", "תחושת שותפות ומעורבות במפגשים", "תרומת הקהילה להתפתחות המקצועית" ], "מידה רבה מאוד / רבה (%)": [92.3, 92.3, 84.6, 76.9], "מידה בינונית (%)": [7.7, 0.0, 15.4, 23.1], "מידה מועטה / כלל לא (%)": [0.0, 7.7, 0.0, 0.0] } df_quant = pd.DataFrame(data_quantitative) # קוביות מדדים עליונות (KPIs) col1, col2, col3, col4 = st.columns(4) with col1: st.metric(label="שביעות רצון כללית (רבה)", value="92.3%", delta="יעד הושג") with col2: st.metric(label="רלוונטיות התכנים (רבה)", value="92.3%", delta="גבוה מאוד") with col3: st.metric(label="מעורבות ושותפות במפגשים", value="84.6%") with col4: st.metric(label="תרומה להתפתחות המקצועית", value="76.9%") st.markdown("---") # חלק א': נתונים כמותיים וגרפים st.header("📈 נתונים כמותיים") # בחירת מדד להצגה בגרף דינמי selected_metric = st.selectbox("בחר מדד לצפייה בגרף התפלגות:", df_quant["מדד הערכה"]) # סינון הנתון שנבחר metric_row = df_quant[df_quant["מדד הערכה"] == selected_metric].iloc[0] chart_data = pd.DataFrame({ "רמת שביעות רצון": ["מידה רבה מאוד / רבה", "מידה בינונית", "מידה מועטה / כלל לא"], "אחוזים": [metric_row["מידה רבה מאוד / רבה (%)"], metric_row["מידה בינונית (%)"], metric_row["מידה מועטה / כלל לא (%)"]] }) # יצירת גרף פאי אינטראקטיבי באמצעות Plotly fig_pie = px.pie(chart_data, values="אחוזים", names="רמת שביעות רצון", title=f"התפלגות התשובות עבור: {selected_metric}", color_discrete_sequence=px.colors.sequential.RdBu) fig_pie.update_traces(textposition='inside', textinfo='percent+label') st.plotly_chart(fig_pie, use_container_width=True) # הצגת טבלת הנתונים המלאה st.subheader("טבלת ריכוז נתונים כמותיים מלאה") st.dataframe(df_quant, use_container_width=True) st.markdown("---") # חלק ב': ממצאים איכותניים ותחומי השפעה st.header("💡 תובנות איכותניות ומגמות מהשדה") tab1, tab2, tab3 = st.tabs(["📌 נקודות חוזק (לשימור)", "🎯 אתגרים והמלצות (לשיפור)", "🎙️ מהשטח (ציטוטים ומנחות)"]) with tab1: st.subheader("תחומי השפעה וחוזקות מרכזיים") st.markdown(""" * **עבודה מבוססת נתונים (Data-Driven):** הרכזים רכשו כלים מעשיים לניתוח מבדקי ראמ"ה פנימיים וחיצוניים, סקרי אקלים ותוכניות 'תנופה', ותרגומם לגרפים וטבלאות עבודה בית-ספריות. * **חדשנות טכנולוגית:** שילוב מוצלח של כלי בינה מלאכותית (AI) ומערכות ניהול דיגיטליות (כלנית, שאלוני מילגו, נוטבוק). * **הפגת הבדידות בתפקיד:** יצירת מרחב בטוח ולמידת עמיתים מעשירה שאיפשרה שיתוף דילמות ואתגרים משותפים. """) with tab2: st.subheader("המלצות ותוכנית עבודה לשנת הלימודים הבאה") st.markdown(""" 1. **הרחבת הזמן היישומי:** דרישה מהשטח ליותר התנסות מעשית 'על רטוב' במערכות המחשוב והמיפוי של ראמ"ה במהלך המפגשים. 2. **דיפרנציאליות לוותק:** יצירת תכנים מתקדמים ומסלול צמיחה מותאם לרכזים ותיקים (קפיצת מדרגה מקצועית). 3. **דיוק הגדרת התפקיד:** הגדרת גבולות גזרה אסטרטגיים מול מנהלי בתי הספר כדי למנוע עומס ופיזור על פני יותר מדי משימות הערכה. """) with tab3: st.subheader("שבחי ההנחיה והערכת המשתתפים") st.markdown(""" * **איכות המנחות:** המשתתפים ציינו לטובה את המקצועיות, הנגשת החומר המורכב 'בגובה העיניים', הזמינות הגבוהה והביקורים הפיזיים בשטח (כולל שיחות משולשות מול הנהלות בתי הספר). * **תרבות הערכה מצמיחה:** הלמידה בקהילה שינתה את התפיסה הבית-ספרית – שימוש בנתונים לצמיחה ופיתוח, ולא ככלי ביקורתי או מעניש. """) st.markdown("---") st.info("💡 הדשבורד נבנה עבור מחוז רכזי הערכה ומוכן להצגה במפגשי פיקוח והנהלה.")
